```python
import requests
from bs4 import BeautifulSoup

def scrape_stock_data(stock_symbol):
    url = f"https://finance.yahoo.com/quote/{stock_symbol}"
    response = requests.get(url)

    if response.status_code == 200:
        soup = BeautifulSoup(response.text, "html.parser")

        stock_data = {
            "symbol": stock_symbol,
            "price": None,
            "previous_close": None,
            "open": None,
            "volume": None,
            "market_cap": None,
            "pe_ratio": None,
            "dividend_yield": None
        }

        try:
            stock_data["price"] = float(soup.find("span", class_="Trsdu(0.3s) Fw(b) Fz(36px) Mb(-4px) D(ib)").text)
            financial_info = soup.find("table", class_="W(100%)")
            rows = financial_info.find_all("tr")

            for row in rows:
                key = row.find("span").text.strip()
                value = row.find("td", class_="Ta(end)").text.strip()

                if key == "Previous Close":
                    stock_data["previous_close"] = float(value)
                elif key == "Open":
                    stock_data["open"] = float(value)
                elif key == "Volume":
                    stock_data["volume"] = int(value.replace(",", ""))
                elif key == "Market Cap":
                    stock_data["market_cap"] = value
                elif key == "PE Ratio (TTM)":
                    stock_data["pe_ratio"] = float(value)
                elif key == "Dividend Yield":
                    stock_data["dividend_yield"] = value

        except Exception as e:
            print(f"Error processing stock data for {stock_symbol}: {e}")

        return stock_data
    else:
        print(f"Error retrieving stock data for {stock_symbol}: {response.status_code}")
        return None

if __name__ == "__main__":
    stock_symbol = "AAPL"
    stock_data = scrape_stock_data(stock_symbol)
    print(stock_data)
```
