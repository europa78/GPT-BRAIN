  <script type="module">
    /**
     * Modern JS:
     * - module script
     * - event delegation
     * - IntersectionObserver reveal + scroll spy
     * - native <dialog> modals
     * - Constraint Validation API for forms
     */

    const $ = (sel, root = document) => root.querySelector(sel);
    const $ = (sel, root = document) => [...root.querySelectorAll(sel)];

    // ===== Customize these =====
    const PHONE = "(your phone)"; // e.g., "(954) 555-1234"
    $("#phoneInline").textContent = `Call: ${PHONE}`;
    $("#year").textContent = String(new Date().getFullYear());

    // Sticky header shadow
    const header = $("[data-header]");
    const onScroll = () => header.classList.toggle("is-scrolled", window.scrollY > 8);
    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });

    // Toast helper
    const toast = $("#toast");
    const toastMsg = $("#toastMsg");
    let toastTimer;
    const showToast = (msg) => {
      toastMsg.textContent = msg;
      toast.dataset.show = "true";
      clearTimeout(toastTimer);
      toastTimer = setTimeout(() => (toast.dataset.show = "false"), 1800);
    };

    // Drawer helpers
    const drawer = $("[data-drawer]");
    const drawerSub = $("[data-drawer-sub]");
    const setDrawer = (open) => {
      drawer.dataset.open = open ? "true" : "false";
      drawer.setAttribute("aria-hidden", open ? "false" : "true");
      document.body.style.overflow = open ? "hidden" : "";
    };

    // Dialog helpers
    const dlgRequest = $("#dlgRequest");
    const dlgLightbox = $("#dlgLightbox");

    const openDialog = (dlg) => {
      if (!dlg) return;
      dlg.showModal?.();
      document.body.style.overflow = "hidden";
    };
    const closeDialog = (dlg) => {
      if (!dlg) return;
      dlg.close?.();
      document.body.style.overflow = "";
    };

    // Click-outside close for dialogs
    [dlgRequest, dlgLightbox].forEach(dlg => {
      dlg?.addEventListener("click", (e) => {
        const r = dlg.getBoundingClientRect();
        const inDialog =
          r.top <= e.clientY && e.clientY <= r.bottom &&
          r.left <= e.clientX && e.clientX <= r.right;
        if (!inDialog) closeDialog(dlg);
      });
      dlg?.addEventListener("close", () => { document.body.style.overflow = ""; });
    });

    // Reveal on scroll
    const revealEls = $("[data-reveal]");
    const ioReveal = new IntersectionObserver((entries) => {
      for (const ent of entries) {
        if (!ent.isIntersecting) continue;
        ent.target.classList.add("is-visible");
        ioReveal.unobserve(ent.target);
      }
    }, { threshold: 0.14 });
    revealEls.forEach(el => ioReveal.observe(el));

    // Scroll spy
    const navLinks = $("[data-nav]").filter(a => (a.getAttribute("href") || "").startsWith("#"));
    const primaryIds = ["home","services","about","photos","contact"];
    const sections = primaryIds.map(id => document.getElementById(id)).filter(Boolean);

    const setActiveNav = (id) => {
      const hash = `#${id}`;
      for (const a of navLinks) {
        const match = a.getAttribute("href") === hash;
        a.classList.toggle("is-active", match);
        if (match) a.setAttribute("aria-current","page");
        else a.removeAttribute("aria-current");
      }
    };

    const ioSpy = new IntersectionObserver((entries) => {
      for (const ent of entries) {
        if (ent.isIntersecting) setActiveNav(ent.target.id);
      }
    }, { rootMargin: "-60% 0px -35% 0px", threshold: 0.01 });
    sections.forEach(s => ioSpy.observe(s));

    // Service preference
    const KEY = "mrhood_service";
    const rememberService = (v) => v && localStorage.setItem(KEY, v);
    const recallService = () => localStorage.getItem(KEY) || "";

    // Prefill remembered service (contact form)
    const serviceSelect = $("#service");
    if (serviceSelect && !serviceSelect.value) serviceSelect.value = recallService();

    // ===== Event delegation (click) =====
    document.addEventListener("click", async (e) => {
      const t = e.target;

      // Close drawer by clicking overlay
      if (t === drawer && drawer.dataset.open === "true") setDrawer(false);

      // Open items
      const open = t.closest("[data-open]")?.dataset.open;
      if (open === "drawer") setDrawer(true);

      if (open === "request") {
        const svc = t.closest("[data-service]")?.dataset.service || t.dataset.service || "";
        const rqService = $("#rqService");
        if (rqService) {
          if (svc) { rqService.value = svc; rememberService(svc); }
          else if (!rqService.value) rqService.value = recallService();
        }
        openDialog(dlgRequest);
      }

      // Close items
      const close = t.closest("[data-close]")?.dataset.close;
      if (close === "drawer") setDrawer(false);
      if (close === "request") closeDialog(dlgRequest);
      if (close === "lightbox") closeDialog(dlgLightbox);

      // Toggle drawer services
      if (t.closest('[data-toggle="drawer-services"]')) {
        const isOpen = drawerSub.dataset.open === "true";
        drawerSub.dataset.open = isOpen ? "false" : "true";
        t.closest('[data-toggle="drawer-services"]').setAttribute("aria-expanded", String(!isOpen));
      }

      // Dropdown (desktop)
      if (t.closest("[data-dropdown-btn]")) {
        const dd = t.closest("[data-dropdown]");
        const isOpen = dd.dataset.open === "true";
        dd.dataset.open = isOpen ? "false" : "true";
        t.closest("[data-dropdown-btn]").setAttribute("aria-expanded", String(!isOpen));
      } else {
        const dd = $("[data-dropdown]");
        if (dd) {
          dd.dataset.open = "false";
          $("[data-dropdown-btn]")?.setAttribute("aria-expanded","false");
        }
      }

      // Flip cards
      if (t.closest('[data-action="flip"]')) {
        const card = t.closest("[data-flip]");
