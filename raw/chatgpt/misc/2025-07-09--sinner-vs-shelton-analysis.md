```python
import os, asyncio, aiohttp, json, time

API_KEY = os.getenv("ODDS_API_KEY")
SPORT   = "tennis_wimbledon"
REGIONS = "us,uk"
ODDS_URL = f"https://api.the-odds-api.com/v4/sports/{SPORT}/odds"

async def fetch_odds(session):
    params = {"regions": REGIONS, "oddsFormat": "decimal", "apiKey": API_KEY}
    async with session.get(ODDS_URL, params=params) as resp:
        payload = await resp.json()
        ts = time.time()
        for event in payload:
            for book in event["bookmakers"]:
                for market in book["markets"]:
                    if market["key"] == "h2h":          # money-line
                        row = {
                            "ts": ts,
                            "match_id": event["id"],
                            "player": market["outcomes"][0]["name"],
                            "price": market["outcomes"][0]["price"]
                        }
                        print(json.dumps(row))          # ↪ push to Kafka
async def main():
    async with aiohttp.ClientSession() as sess:
        while True:
            await fetch_odds(sess)
            await asyncio.sleep(10)                    # respect rate-limit
asyncio.run(main())
```
