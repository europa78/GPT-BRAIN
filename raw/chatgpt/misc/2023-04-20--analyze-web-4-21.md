```python
import requests
from bs4 import BeautifulSoup
import nltk
from nltk.corpus import stopwords
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation
from wordcloud import WordCloud
import matplotlib.pyplot as plt

# Download NLTK stopwords if not already downloaded
nltk.download('stopwords')


def fetch_content(url):
    try:
        response = requests.get(url)
        response.raise_for_status()
        soup = BeautifulSoup(response.text, 'html.parser')
        content = ' '.join([p.text for p in soup.find_all('p')])
        return content
    except requests.exceptions.RequestException as e:
        print(f"Error fetching content from {url}: {e}")
        return ""


def preprocess_text(texts):
    stop_words = stopwords.words('english')
    vectorizer = TfidfVectorizer(stop_words=stop_words)
    return vectorizer.fit_transform(texts)


def get_topics(model, vectorizer, top_n=10):
    for idx, topic in enumerate(model.components_):
        print(f"Topic {idx + 1}:")
        print(" ".join([vectorizer.get_feature_names()[i] for i in topic.argsort()[:-top_n - 1:-1]]))


def plot_wordcloud(model, vectorizer, topic_idx, n_words=100, save_as_image=False, image_filename=None):
    topic = model.components_[topic_idx]
    wordcloud = WordCloud(
        background_color='white',
        max_words=n_words,
    )
    wordcloud.generate_from_frequencies(zip(vectorizer.get_feature_names(), topic))
    plt.imshow(wordcloud, interpolation='bilinear')
    plt.axis("off")
    plt.show()

    if save_as_image and image_filename:
        wordcloud.to_file(image_filename)


# Replace this list with your 20 URLs
urls = [
    'https://example1.com',
    'https://example2.com',
    # ...
]

# Fetch content from URLs
texts = [fetch_content(url) for url in urls]

# Remove empty content (if any errors occurred)
texts = [text for text in texts if text]

# Preprocess texts
X = preprocess_text(texts)

# Perform LDA
lda = LatentDirichletAllocation(n_components=5, random_state=42)
lda.fit(X)

# Get and display topics
vectorizer = preprocess_text(texts)
get_topics(lda, vectorizer)

# Plot wordcloud for a specific topic (and save as an image)
plot_wordcloud(lda, vectorizer, topic_idx=0, save_as_image=True, image_filename="wordcloud_topic_1.png")
```

- Error handling for `requests.get()` in the `fetch_content()` function.
- Removal of empty content from `texts` if any errors occurred during content fetching.
- Added the `save_as_image` and `image_filename` parameters to the `plot_wordcloud()` function to save word clouds as images.
- Removed unnecessary imports (pandas, numpy, seaborn) and reduced the number of imports by using the functions directly.
