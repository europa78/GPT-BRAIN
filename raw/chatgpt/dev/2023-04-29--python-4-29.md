def build_entity_network(articles):
    G = nx.Graph()

    for article in articles:
        doc = nlp(article)
        entities = [ent.text for ent in doc.ents if ent.label_ in ["ORG", "PERSON", "GPE"]]

        for entity in entities:

        for i, source in enumerate(entities):
            for target in entities[i + 1:]:
                if G.has_edge(source, target):
                    G[source][target]["weight"] += 1
                else:
                    G.add_edge(source, target, weight=1)

    return G

nlp = spacy.load("en_core_web_sm")

# Initialize the BERT tokenizer and model
tokenizer = BertTokenizer.from_pretrained("bert-large-uncased")
model = BertModel.from_pretrained("bert-large-uncased")

def analyze_url(url):
    # Web scraping
    response = requests.get(url)
    soup = BeautifulSoup(response.text, "html.parser")
    text = " ".join([p.text for p in soup.find_all("p")])

    # Keyword extraction
    r = Rake()
    r.extract_keywords_from_text(text)
    keywords = r.get_ranked_phrases()

    # Sentiment analysis
    doc = nlp(text)
    sentiment = doc.sentiment

    # Text summarization
    inputs = tokenizer(text, return_tensors="pt", truncation=True, padding=True)
    outputs = model(**inputs)
    last_hidden_state = outputs.last_hidden_state
    cls_vector = last_hidden_state[:, 0, :]
    most_similar_sentence_indices = np.argsort(np.dot(cls_vector.detach().numpy(), last_hidden_state.detach().numpy().T))[0, -2:]
    summary = " ".join([doc[sent.start: sent.end].text for sent in doc.sents if doc[sent.start].i in most_similar_sentence_indices])

    # Network analysis
    G = nx.Graph()
    for sent in doc.sents:
        for token1 in sent:
            for token2 in sent:
                if token1 != token2:
                    G.add_edge(token1.text, token2.text)

    return {
        "keywords": keywords,
        "sentiment": sentiment,
        "summary": summary,
        "network": G,
    }

url = "https://en.wikipedia.org/wiki/Web_scraping"
